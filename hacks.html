Background Gradient

makes a smooth color changing background on most pages

let x = 0; function smooth(){ document.body.style.backgroundColor = "rgb(" + Math.cos(x)*255 + "," + Math.cos(x + 2.1)*255 + "," + Math.cos(x + 4.2)*255 + ")"; x = x + 0.01; document.getElementById("demo").innerHTML = x; } setInterval(smooth, 75);

Trip

this gets a random element on the screen that is visible and changes its color and background colors to something random every millisecond

trip(); function trip(){ let all = document.all; let max = all.length; let edit = all[Math.floor(Math.random()*max)]; var stopat = 1000; var count = 0; while(edit.getBoundingClientRect().width == 0){ edit = all[Math.floor(Math.random()*max)]; count++; if(count > stopat){ return; } } let r = Math.floor(Math.random()*255); let g = Math.floor(Math.random()*255); let b = Math.floor(Math.random()*255); edit.style.color = "rgb(" + r + ", " + g + ", " + b + ")"; r = Math.floor(Math.random()*255); g = Math.floor(Math.random()*255); b = Math.floor(Math.random()*255); edit.style.backgroundColor = "rgb(" + r + ", " + g + ", " + b + ")"; setTimeout(trip, 1); }
